{\bf Questão 2: gabarito do item a}

\def\eqa{\overset{\ColorRed{?}}{=}}

$$\begin{array}{l}
    \intx {e^{x^5} x^4} \\
    = \;\; \intu {e^u · \frac{1}{5}} \\
    = \;\; \frac{1}{5} \intu {e^u} \\
    = \;\; \frac{1}{5} e^u \\
    = \;\; \frac{1}{5} e^{x^5} \\
    \end{array}
  \quad
  \bmat{ u = x^5 \\
         \frac{du}{dx} = 5 x^4 \\
         du = 5 x^4 \, dx \\
         \frac{1}{5} du = x^4 \, dx \\
       }
  \qquad
  \begin{array}{rcl}
  \intx {e^{x^5} x^4} &\eqa& \ph{m,}         \frac{1}{5} e^{x^5} \\
         e^{x^5} x^4  \ph{mi} &\eqa& \ddx \, \frac{1}{5} e^{x^5} \\
                                 &=&  \frac{1}{5} e^{x^5} 5 x^4 \\
                                 &=&   e^{x^5} x^4 \\
  \end{array}
$$

{\bf Questão 2: gabarito do item b}

$$\begin{array}{l}
    \intx {\sqrt{2 + \sen x} · \cos x} \\
    = \;\; \intu {\sqrt{u}} \\
    = \;\; \intu {u^{\frac{1}{2}}} \\
    = \;\; \frac{2}{3} u^{\frac{3}{2}} \\
    = \;\; \frac{2}{3} (2+\sen x)^{\frac{3}{2}} \\
    \end{array}
  \quad
  \bmat{ u = 2 + \sen x \\
         \frac{du}{dx} = \cos x \\
         du = \cos x \, dx \\
       }
$$

\bsk

$$\begin{array}{l}
    \ddx \left( \frac{2}{3} (2+\sen x)^{\frac{3}{2}} \right) \\
    = \;\; \frac{2}{3} · \ddx (2+\sen x)^{\frac{3}{2}}  \\[2.5pt]
